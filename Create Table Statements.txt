CREATE TABLE IF NOT EXISTS User(
	UserID INTEGER PRIMARY KEY,
	Username VARCHAR(50) NOT NULL,
	Password VARCHAR(50) NOT NULL,
	EmailAddress VARCHAR(75) NOT NULL,
	State VARCHAR(2) NOT NULL,
	City VARCHAR(50) NOT NULL,
	Street VARCHAR(75) NOT NULL,
	Apartment VARCHAR(10),
	Zip VARCHAR(5) NOT NULL
);

CREATE TABLE IF NOT EXISTS AuthToken(
	UserID INTEGER PRIMARY KEY,
	AuthToken VARCHAR(50) NOT NULL,
	FOREIGN KEY (UserID) REFERENCES Users(UserID)
		ON DELETE NO ACTION
);

CREATE TABLE IF NOT EXISTS TerrariumModel(
	ModelID INTEGER PRIMARY KEY,
	SpaceAvailable REAL NOT NULL
);

CREATE TABLE IF NOT EXISTS Plant(
	PlantID INTEGER PRIMARY KEY,
	Name VARCHAR(50) NOT NULL,
	Temperature REAL NOT NULL,
	SoilMoisture REAL NOT NULL,
	Humidity REAL NOT NULL,
	LightLevel REAL NOT NULL,
	GrowthTimeDays INTEGER NOT NULL,
	SpaceRequirement REAL NOT NULL
);

CREATE TABLE IF NOT EXISTS LiveTerrarium(
	TerrariumID INTEGER PRIMARY KEY,
	UserID INTEGER NOT NULL,
	ModelID INTEGER NOT NULL,
	CurrentPlantID INTEGER,
	Temperature REAL,
	SoilMoisture REAL,
	Humidity REAL,
	LightLevel REAL,
	DaysGrown INTEGER,
	FOREIGN KEY (ModelID)
		REFERENCES TerrariumModel(ModelID)
			ON DELETE NO ACTION,
	FOREIGN KEY (UserID)
		REFERENCES User(UserID)
			ON DELETE NO ACTION,
	FOREIGN KEY (CurrentPlantID)
		REFERENCES Plant(PlantID)
			ON DELETE NO ACTION
);
